To show

new Modal(button, { content: '.modal_1' }).open();

To hide

new Modal(b1, { content: '.modal_1' }).close();

          </div> 
          <div class="modal-body">
            ...
          </div> 
          <div class="modal-footer">
            <button class="close-modal">Close</button>
            <button class="confirm-modal">OK</button>
          </div> 
        </div>
